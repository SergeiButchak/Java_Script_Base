        class ShoppingCartItem {
            constructor(itemId, count, price) {
                this.itemId = itemId;
                this.count = count;
                this.price = price;
            }

            getTotal() {
                return this.count * this.price;
            }
        }   
        
        function countBasketPrice(cart) {
            let res = 0;
            for (let i = 0; i < cart.length; i++) {
                res += cart[i].getTotal();
            }   
            return res;     
        }

        let cart = [];
        cart.push(new ShoppingCartItem(1, 5, 154));
        cart.push(new ShoppingCartItem(5, 1, 1500));
        cart.push(new ShoppingCartItem(51, 15, 67));
        cart.push(new ShoppingCartItem(47, 7, 245));
        cart.push(new ShoppingCartItem(99, 2, 2000));
        document.write(countBasketPrice(cart));
