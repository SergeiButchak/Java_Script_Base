1. С помощью цикла while вывести все простые числа в промежутке от 0 до 100.

        let number = 2;
        while (number <= 100) {
            let num = 2;
            let res = true;
            while (num < number) {
                if (number % num == 0) {
                    res = false;
                    break;
                }
                num++;
            }
            if (res)
                document.write(`${number} `);
            number++;
        }
