        // для хранени данных о товаре будем использовать массив из трех элементов, хранящих "название", "стоимость" и "количество"
        function countBasketPrice(cart) {
            res = 0;
            for (i = 0; i < cart.length; i++)
                res += cart[i][1] * cart[i][2];
            
            return res;
