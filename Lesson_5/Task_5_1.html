Практическое задание 1

1. Создать функцию, генерирующую шахматную доску. Можно использовать любые html-теги. Доска должна быть верно разлинована на черные и белые ячейки. Строки должны нумероваться числами от 1 до 8, столбцы — латинскими буквами A, B, C, D, E, F, G, H. 

        <style>
            .row {
                display: flex;
            }
            .square {
                width: 100px;
                height: 100px;
                font-size: 24px;
                text-align: center;
                line-height: 100px;
                border: 1px solid white;
            }
    
            .cell {
                border: 1px solid black;
                border-collapse: collapse;
            }
            .even>.cell:nth-child(odd) {
                background-color: black;
            }
    
            .even>.cell:nth-child(even) {
                background-color: orange;
            }
            .odd>.cell:nth-child(even) {
                background-color: black;
            }
    
            .odd>.cell:nth-child(odd) {
                background-color: orange;
            }
    
        </style>
        <script>
            let desk = [
                ['', 'A', 'B', 'C', 'D', 'E', 'F', 'G', 'H'],
                ['8', '', '', '', '', '', '', '', ''],
                ['7', '', '', '', '', '', '', '', ''],
                ['6', '', '', '', '', '', '', '', ''],
                ['5', '', '', '', '', '', '', '', ''],
                ['4', '', '', '', '', '', '', '', ''],
                ['3', '', '', '', '', '', '', '', ''],
                ['2', '', '', '', '', '', '', '', ''],
                ['1', '', '', '', '', '', '', '', ''],
            ];
            let parity_row = "";
            for (let i = 0; i <= 8; i++){
                let row = document.createElement("div");
                let row_class = "row" + parity_row;
                row.setAttribute("class", row_class);
                let parity_cell = "";
                for (let j = 0; j <= 8; j++){
                    let cell = document.createElement("div");
                    let cell_class = "square" + parity_cell;
                    cell.setAttribute("class", cell_class);
                    cell.innerHTML = desk[i][j];
                    row.appendChild(cell);
                    if (i > 0)
                        parity_cell = " cell";
                }
                document.body.appendChild(row);
                if (i % 2 == 0) {
                    parity_row = " even";
                }
                else {
                    parity_row = " odd";
                }
    
            }
        </script>
